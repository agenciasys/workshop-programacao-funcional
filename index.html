Conceito

First-Class Functions

Funções atribuidas a variáveis. 

$showMessage = function($message) {
	echo $message;
};

var showMessage = function(message) {
	console.log(message);
};

High-Order Functions

Funções que recebem ou retornam outras funções. 

$isOver18 = function($person) {
    return $person['age'] >= 18;
};

$output = array_filter($people, $isOver18);

var isOver18 = function(person) {
    return +person.age >= 18;
},

output = _.filter(people, isOver18);

Closures/ funções anônimas/ lambda

$output = array_filter($people, function($person) {
    return $person['age'] >= 18;
});

var output = _.filter(people, function(person) {
    return +person.age >= 18;
});

Imutabilidade

$original = [1, 2, 3, 4, 5];
$new = array_map(function($item) {
	return $item * 2;
}, $original);

// $original retorna [1, 2, 3, 4, 5]
// $new retorna [2, 4, 6, 8 10]
